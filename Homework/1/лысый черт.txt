import java.util.Random;
import java.util.ArrayList;
import java.util.List;

// ========== ПЕРЕЧИСЛЕНИЯ ==========
enum Race {
    HUMAN("Человек"), ORC("Орк"), ELF("Эльф"), DEMON("Лысый черт");
    private final String name;
    Race(String name) { this.name = name; }
    public String getName() { return name; }
}

enum WeaponType {
    SWORD("Меч", 10, 5), AXE("Топор Великого не повторомивого велечайщего самого крутого прям вообще крутой как сам ИванЗоло2004", 15, -5), BOW("Лук", 8, 15),
    BEER_BOTTLE("Пивная бутылка", 20, -10), LENIVETS_CLAW("Пиво темное", 25, 0);
    private final String name;
    private final int damageBonus;
    private final int accuracyBonus;
    WeaponType(String name, int dmg, int acc) { this.name = name; this.damageBonus = dmg; this.accuracyBonus = acc; }
    public String getName() { return name; }
    public int getDamageBonus() { return damageBonus; }
    public int getAccuracyBonus() { return accuracyBonus; }
}

enum ArmorType {
    RAGS("Лохмотья", 0, 0), LEATHER("Geynal", 5, -2),
    CHAINMAIL("Кольчуга", 10, -5), PLATE("Латы", 15, -10),
    LYSINA("Лысина Владимира", 50, 20); // Имба-броня

    private final String name;
    private final int defense;
    private final int dodgePenalty;
    ArmorType(String name, int def, int pen) { this.name = name; this.defense = def; this.dodgePenalty = pen; }
    public String getName() { return name; }
    public int getDefense() { return defense; }
    public int getDodgePenalty() { return dodgePenalty; }
}

// ========== ПРЕДМЕТЫ ==========
class Weapon {
    private final WeaponType type;
    public Weapon(WeaponType type) { this.type = type; }
    public int getDamageBonus() { return type.getDamageBonus(); }
    public int getAccuracyBonus() { return type.getAccuracyBonus(); }
    public String getName() { return type.getName(); }
}

class Armor {
    private final ArmorType type;
    public Armor(ArmorType type) { this.type = type; }
    public int getDefense() { return type.getDefense(); }
    public int getDodgePenalty() { return type.getDodgePenalty(); }
    public String getName() { return type.getName(); }
}

// ========== БАЗОВЫЙ АБСТРАКТНЫЙ КЛАСС ГЕРОЯ ==========
abstract class Hero {
    protected String name;
    protected Race race;
    protected int health;
    protected int maxHealth;
    protected int baseDamage;
    protected int baseAccuracy;
    protected Weapon weapon;
    protected Armor armor;
    protected Random rand = new Random();
    protected Vyacheslav host = Vyacheslav.getInstance();

    public Hero(String name, Race race, int health, int damage, int accuracy) {
        this.name = name;
        this.race = race;
        this.maxHealth = health;
        this.health = health;
        this.baseDamage = damage;
        this.baseAccuracy = accuracy;
        this.weapon = new Weapon(WeaponType.SWORD); // По умолчанию
        this.armor = new Armor(ArmorType.RAGS);     // По умолчанию
    }

    public void equipWeapon(Weapon w) { this.weapon = w; }
    public void equipArmor(Armor a) { this.armor = a; }

    public boolean isAlive() { return health > 0; }
    public String getName() { return name; }
    public int getHealth() { return health; }
    public int getMaxHealth() { return maxHealth; }

    public void takeDamage(int dmg) {
        int reduced = Math.max(1, dmg - armor.getDefense());
        health -= reduced;
        if (health < 0) health = 0;
    }

    public void reset() { health = maxHealth; }
    public void heal(int amount) { health = Math.min(maxHealth, health + amount); }

    // Обычная атака с шансом промаха
    public boolean attack(Hero enemy) {
        int totalAccuracy = baseAccuracy + weapon.getAccuracyBonus() - enemy.armor.getDodgePenalty();
        int hit = rand.nextInt(100);

        if (hit < totalAccuracy) {
            int dmg = baseDamage + weapon.getDamageBonus() + rand.nextInt(10);
            enemy.takeDamage(dmg);
            host.commentAttack(this, enemy, dmg, true);
            return true;
        } else {
            host.commentAttack(this, enemy, 0, false);
            return false;
        }
    }

    // Рофленая способность (у каждой расы своя)
    public abstract void useRoflAbility(Hero enemy);

    public String getFullDescription() {
        return name + " [" + race.getName() + "] | Оружие: " + weapon.getName() +
                " | Броня: " + armor.getName() + " | HP: " + health + "/" + maxHealth;
    }
}

// ---------- ОРК (тупой, но сильный) ----------
class Orc extends Hero {
    public Orc(String name) {
        super(name, Race.ORC, 120, 20, 60);
        equipWeapon(new Weapon(WeaponType.AXE));
        equipArmor(new Armor(ArmorType.LEATHER));
    }

    @Override
    public void useRoflAbility(Hero enemy) {
        int choice = rand.nextInt(3);
        switch (choice) {
            case 0:
                host.commentAction(name + " ПЫТАЕТСЯ ДУМАТЬ... НО НЕ МОЖЕТ! Сам себе наносит 5 урона от перенапряжения!");
                takeDamage(5);
                break;
            case 1:
                int dmg = baseDamage + 15;
                enemy.takeDamage(dmg);
                takeDamage(8);
                host.commentAction(name + " ВПАДАЕТ В БЕРСЕРК! Наносит " + dmg + " урона " + enemy.getName() + ", но бьётся головой об стену (-8 HP)!");
                break;
            case 2:
                host.commentAction(name + " ЖРЁТ ГРИБЫ?... " + (rand.nextBoolean() ? "ЛЕЧЕБНЫЕ! +15 HP!" : "ЯДОВИТЫЕ! -10 HP!"));
                if (rand.nextBoolean()) heal(15); else takeDamage(10);
                break;
        }
    }
}

// ========== АРЕНА РОССИЯ 23 ==========
class Russia23Arena {
    private Vyacheslav host = Vyacheslav.getInstance();
    private List<Hero> fighters = new ArrayList<>();
    private Random rand = new Random();

    public void addFighter(Hero hero) { fighters.add(hero); }

    public void startTournament() {
        host.startShow();

        // РАУНД 1: Орк vs Эльф (до 3 ходов)
        Hero orc = fighters.stream().filter(h -> h instanceof Orc).findFirst().orElse(null);
        Hero elf = fighters.stream().filter(h -> h instanceof Elf).findFirst().orElse(null);

        System.out.println("=== РАУНД 1: ОРК vs ЭЛЬФ ===");
        host.announceFighters(orc, elf);
        Hero winner1 = fight(orc, elf, 3);
        host.announceWinner(winner1);
        winner1.heal(30);

        // РАУНД 2: Победитель vs Владимир (до 3 ходов)
        Hero vladimir = fighters.stream().filter(h -> h instanceof Vladimir).findFirst().orElse(null);
        vladimir.reset();

        System.out.println("=== РАУНД 2: ФИНАЛ ЗА НИЧЕГО ===");
        host.announceFighters(vladimir, winner1);
        Hero winner2 = fight(vladimir, winner1, 3);
        host.announceWinner(winner2);

        // ФИНАЛ
        host.screenGlitch();
        host.finalScene();
        host.commercial();
    }

    private Hero fight(Hero h1, Hero h2, int maxTurns) {
        for (int turn = 1; turn <= maxTurns; turn++) {
            System.out.println("\n--- Ход " + turn + " ---");

            // 50% шанс использовать способность
            if (rand.nextBoolean()) {
                h1.useRoflAbility(h2);
            } else {
                h1.attack(h2);
            }
            host.showHP(h1, h2);
            if (!h2.isAlive()) return h1;

            if (rand.nextBoolean()) {
                h2.useRoflAbility(h1);
            } else {
                h2.attack(h1);
            }
            host.showHP(h1, h2);
            if (!h1.isAlive()) return h2;
        }
        return h1.getHealth() >= h2.getHealth() ? h1 : h2;
    }
}

        arena.addFighter(new Vladimir());
